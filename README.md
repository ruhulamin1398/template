#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
        ll n;
        cin>>n;
        string s1,s2;
        cin>>s1;
        cin>>s2;
        ll flag=0,x=1;
        for(ll i=0; i<n; i++)
        {
            if(x==1 and s1[i]>'2'){
                if(s2[i]>'2'){
                    x=2;
                }
                else
                {

                    flag=1;
                    break;
                }
                continue;
            }
            if(x==2 and s2[i]>'2'){
                if(s1[i]>'2'){
                    x=1;
                }
                else
                {
                    flag=1;
                    break;
                }
                continue;
            }


        }
        if(flag ==1 || x==1)
            cout<<"NO"<<endl;
        else
            cout<<"YES"<<endl;
    }



    return 0;
}
